19#define BLOCKSIZE 1024
20#define IV_SIZE 32
21#define KEY_SIZE 32

23extern int do_crypt(FILE* in, FILE* out, int action, unsigned char *key_str){
24 /* Local Vars */
25
26 /* Buffers */
27 unsigned char inbuf[BLOCKSIZE];
28 int inlen;
29 /* Allow enough space in output buffer for additional cipher block */
30 unsigned char outbuf[BLOCKSIZE + EVP_MAX_BLOCK_LENGTH];
31 int outlen;
32 int writelen;
33
34 /* OpenSSL libcrypto vars */
35 EVP_CIPHER_CTX *ctx;
36 ctx = EVP_CIPHER_CTX_new();
37
38 unsigned char key[KEY_SIZE];
39 unsigned char iv[IV_SIZE];
40 int nrounds = 5;
41
42 /* tmp vars */
43 int i;
44 /* Setup Encryption Key and Cipher Engine if in cipher mode */
45 if(action >= 0){
46 if(!key_str){
47 /* Error */
48 fprintf(stderr, "Key_str must not be NULL\n");
49 return 0;
50 }
51 /* Build Key from String */
52 i = EVP_BytesToKey(EVP_aes_256_cbc(), EVP_sha1(), NULL,
53 key_str, (int)strlen((const char *)key_str), nrounds, key, iv);
54 if (i != 32) {
55 /* Error */
56 fprintf(stderr, "Key size is %d bits - should be 256 bits\n", i*8);
57 return 0;
58 }
59 /* Init Engine */
60 EVP_CIPHER_CTX_init(ctx);
61 EVP_CipherInit_ex(ctx, EVP_aes_256_cbc(), NULL, key, iv, action);
62 }
63
64 /* Loop through Input File*/
65 for(;;){
66 /* Read Block */
67 inlen = fread(inbuf, sizeof(*inbuf), BLOCKSIZE, in);
68 if(inlen <= 0){
69 /* EOF -> Break Loop */
70 break;
71 }
72
73 /* If in cipher mode, perform cipher transform on block */
74 if(action >= 0){
75 if(!EVP_CipherUpdate(ctx, outbuf, &outlen, inbuf, inlen))
76 {
77 /* Error */
78 EVP_CIPHER_CTX_cleanup(ctx);
79 return 0;
80 }
81 }
82 /* If in pass-through mode. copy block as is */
83 else{
84 memcpy(outbuf, inbuf, inlen);
85 outlen = inlen;
86 }
87
88 /* Write Block */
89 writelen = fwrite(outbuf, sizeof(*outbuf), outlen, out);
90 if(writelen != outlen){
91 /* Error */
92 perror("fwrite error");
93 EVP_CIPHER_CTX_cleanup(ctx);
94 return 0;
95 }
96 }
97
98 /* If in cipher mode, handle necessary padding */
99 if(action >= 0){
100 /* Handle remaining cipher block + padding */
101 if(!EVP_CipherFinal_ex(ctx, outbuf, &outlen))
102 {
103 /* Error */
104 EVP_CIPHER_CTX_cleanup(ctx);
105 return 0;
106 }
107 /* Write remainign cipher block + padding*/
108 fwrite(outbuf, sizeof(*inbuf), outlen, out);
109 EVP_CIPHER_CTX_cleanup(ctx);
110 }
111
112 /* Success */
113 return 1;
114}

116// Verify the entropy
117int check_entropy(void) {
118 FILE *entropy_file = fopen("/proc/sys/kernel/random/entropy_avail", "r");
119 if (entropy_file == NULL) {
120 perror("Err: Cannot open entropy file");
121 return -1;
122 }
123
124 int entropy_value;
125 if (fscanf(entropy_file, "%d", &entropy_value) != 1) {
126 perror("Err: Cannot estimate entropy");
127 fclose(entropy_file);
128 return -1;
129 }
130
131 fclose(entropy_file);
132 return entropy_value;
133}

136void add_entropy(void) {
137 FILE *urandom = fopen("/dev/urandom", "rb");
138 if (urandom == NULL) {
139 perror("Err: Cannot open /dev/urandom");
140 exit(EXIT_FAILURE);
141 }
142
143 unsigned char random_data[32];
144 size_t bytes_read = fread(random_data, 1, sizeof(random_data), urandom);
145 fclose(urandom);
146
147 if (bytes_read != sizeof(random_data)) {
148 fprintf(stderr, "Err: Cannot read data\n");
149 exit(EXIT_FAILURE);
150 }
151
152 // Usa i dati casuali per aggiungere entropia
153 RAND_add(random_data, sizeof(random_data), 0.5); // 0.5 è un peso arbitrario
154
155 fprintf(stdout, "Entropy added successfully!\n");
156}

159void generate_key(unsigned char *destination) {
160 fprintf(stdout, "Generating a new key...\n");
161
162 //Why? Because if we try to create a large number of files there might not be enough random bytes in the system to generate a key
163 for (int i = 0; i < 10; i++) {
164 int entropy = check_entropy();
165 if (entropy < 128) {
166 fprintf(stderr, "WARN: not enough entropy, creating some...\n");
167 add_entropy();
168 }
169
170 if (RAND_bytes(destination, 32) != 1) {
171 fprintf(stderr, "Err: Cannot generate key\n");
172 destination = NULL;
173 }
174
175 if (strlen((const char *)destination) == 32)
176 break;
177 }
178
179 if (is_valid_key(destination) == 0) {
180 fprintf(stderr, "Err: Generated key is inval1d\n");
181 print_aes_key(destination);
182 destination = NULL;
183 }
184}

186unsigned char* encrypt_string(unsigned char* plaintext, const char* key, int *encrypted_key_len) {
187 EVP_CIPHER_CTX* ctx;
188 const EVP_CIPHER* cipher = EVP_aes_256_cbc();
189 unsigned char iv[AES_BLOCK_SIZE];
190 memset(iv, 0, AES_BLOCK_SIZE);
191
192 ctx = EVP_CIPHER_CTX_new();
193 if (!ctx) {
194 return NULL;
195 }
196
197 EVP_EncryptInit_ex(ctx, cipher, NULL, (const unsigned char*)key, iv);
198
199 size_t plaintext_len = strlen((const char*)plaintext);
200 unsigned char ciphertext[plaintext_len + AES_BLOCK_SIZE];
201 memset(ciphertext, 0, sizeof(ciphertext));
202
203 int len;
204 EVP_EncryptUpdate(ctx, ciphertext, &len, plaintext, plaintext_len);
205 EVP_EncryptFinal_ex(ctx, ciphertext + len, &len);
206 EVP_CIPHER_CTX_free(ctx);
207
208 unsigned char* encoded_string = malloc(len * 2 + 1);
209 if (!encoded_string) {
210 return NULL;
211 }
212
213 for (int i = 0; i < len; i++) {
214 sprintf((char*)&encoded_string[i * 2], "%02x", ciphertext[i]);
215 }
216 encoded_string[len * 2] = '\0';
217
218 *encrypted_key_len = len * 2;
219 return encoded_string;
220}

223unsigned char* decrypt_string(unsigned char* ciphertext, const char* key) {
224 EVP_CIPHER_CTX *ctx;
225 const EVP_CIPHER *cipher = EVP_aes_256_cbc(); // Choose the correct algorithm
226 unsigned char iv[AES_BLOCK_SIZE];
227 memset(iv, 0, AES_BLOCK_SIZE);
228
229 ctx = EVP_CIPHER_CTX_new();
230 EVP_DecryptInit_ex(ctx, cipher, NULL, (const unsigned char*)key, iv);
231
232 size_t decoded_len = strlen((const char *)ciphertext);
233
234 unsigned char plaintext[decoded_len];
235 memset(plaintext, 0, sizeof(plaintext));
236
237 int len;
238 EVP_DecryptUpdate(ctx, plaintext, &len, ciphertext, (int )decoded_len);
239 EVP_DecryptFinal_ex(ctx, plaintext + len, &len);
240 EVP_CIPHER_CTX_free(ctx);
241
242 unsigned char* decrypted_string = (unsigned char*)malloc(decoded_len + 1);
243 memcpy(decrypted_string, plaintext, decoded_len);
244 decrypted_string[decoded_len] = '\0';
245
246
247 return decrypted_string;
248}
