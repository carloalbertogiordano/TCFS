195static int
196tcfs_readdir (const char *fuse_path, void *buf, fuse_fill_dir_t filler,
197 off_t offset, struct fuse_file_info *fi,
198 enum fuse_readdir_flags frdf)
199{
200 (void)offset;
201 (void)fi;
202 (void)frdf;
203
204 logMessage ("Called readdir %s\n", fuse_path);
205 const char *enc_path = encrypt_path (fuse_path, password);
206 char *path = prefix_path (enc_path, root_path);
207 logMessage ("readdir on %s\n", path);
208
209 DIR *dp = NULL;
210 struct dirent *de;
211
212 dp = opendir (path);
213 if (dp == NULL)
214 {
215 perror ("Could not open the directory");
216 return -errno;
217 }
218 logMessage ("Dir %s opened\n", path);
219
220 while ((de = readdir (dp)) != NULL)
221 {
222 struct stat st;
223 memset (&st, 0, sizeof (st));
224 st.st_ino = de->d_ino;
225 st.st_mode = de->d_type << 12;
226
227 int filler_res = -1, can_break = false;
228
229 if ((strcmp (de->d_name, ".") == 0 || strcmp (de->d_name, "..") == 0
230 || strcmp (de->d_name, "/") == 0))
231 {
232 logMessage ("ONE\n");
233 filler_res = filler (buf, de->d_name, &st, 0, 0);
234 if (filler_res != 0)
235 {
236 can_break = true;
237 }
238 }
239 else
240 {
241 logMessage (
242 "\tchecking for %s is %s\n", de->d_name,
243 decrypt_file_name_with_hex ((const char *)de->d_name, password));
244
245 const char *dec_dirname = decrypt_path (de->d_name, password);
246 if (dec_dirname == NULL)
247 {
248 perror ("Could not decipher dir name");
249 return -ERR_invalid_enc_dir_name;
250 }
251
252 // We must avoid the initial / of dec_dirname
253 filler_res = filler (buf, dec_dirname + 1, &st, 0, 0);
254 if (filler_res != 0)
255 {
256 can_break = true;
257 }
258 }
259
260 logMessage ("FILLER RES: %d, CAN BREAK %d\n", filler_res, can_break);
261 if (can_break)
262 {
263 logMessage ("Breaking out\n");
264 perror ("readdir error");
265 closedir (dp);
266 return -errno;
267 break;
268 }
269 }
270
271 closedir (dp);
272 return TCFS_SUCCESS;
273}

733static int
734tcfs_read (const char *fuse_path, char *buf, size_t size, off_t offset,
735 struct fuse_file_info *fi)
736{
737 (void)size;
738 (void)fi;
739
740 logMessage ("Calling read\n");
741 FILE *path_ptr, *tmpf;
742 char *path;
743 int res;
744
745 // Retrieve the username
746 char username_buf[1024];
747 size_t username_buf_size = 1024;
748 get_user_name (username_buf, username_buf_size);
749
750 const char *enc_fuse_path = encrypt_path (fuse_path, password);
751 path = prefix_path (enc_fuse_path, root_path);
752 logMessage ("\tread on %s\n", path);
753
754 path_ptr = fopen (path, "r");
755 tmpf = tmpfile ();
756
757 // Get key size
758 char *size_key_char = malloc (sizeof (char) * 20);
759 if (tcfs_getxattr (fuse_path, "user.key_len", size_key_char, 20) == -1)
760 {
761 perror ("Could not get file key size");
762 return -errno;
763 }
764 ssize_t size_key = strtol (size_key_char, NULL, 10);
765
766 // Retrive the file key
767 unsigned char *encrypted_key = malloc ((size_key + 1) * sizeof (char));
768 encrypted_key[size_key] = '\0';
769 if (tcfs_getxattr (fuse_path, "user.key", (char *)encrypted_key, size_key)
770 == -1)
771 {
772 perror ("Could not get encrypted key for file in tcfs_read");
773 return -errno;
774 }
775
776 // Decrypt the file key
777 unsigned char *decrypted_key;
778 decrypted_key = decrypt_string (encrypted_key, password);
779
780 /* Decrypt*/
781 if (do_crypt (path_ptr, tmpf, DECRYPT, decrypted_key) != true)
782 {
783 perror ("Err: do_crypt cannot decrypt file");
784 return -errno;
785 }
786
787 /* Something went terribly wrong if this is the case. */
788 if (path_ptr == NULL || tmpf == NULL)
789 return -errno;
790
791 if (fflush (tmpf) != 0)
792 {
793 perror ("Err: Cannot flush file in read process");
794 return -errno;
795 }
796 if (fseek (tmpf, offset, SEEK_SET) != 0)
797 {
798 perror ("Err: cannot fseek while reading file");
799 return -errno;
800 }
801
802 /* Read our tmpfile into the buffer. */
803 res = fread (buf, 1, file_size (tmpf), tmpf);
804 if (res == -1)
805 {
806 perror ("Err: cannot fread whine in read");
807 res = -errno;
808 }
809
810 fclose (tmpf);
811 fclose (path_ptr);
812 free (encrypted_key);
813 free (decrypted_key);
814 return res;
815}

829static int
830tcfs_write (const char *fuse_path, const char *buf, size_t size, off_t offset,
831 struct fuse_file_info *fi)
832{
833 (void)fi;
834 logMessage ("Called write\n");
835
836 FILE *path_ptr, *tmpf;
837 char *path;
838 int res;
839 int tmpf_descriptor;
840
841 const char *enc_fuse_path = encrypt_path (fuse_path, password);
842 path = prefix_path (enc_fuse_path, root_path);
843 logMessage ("\twrite on %s\n", path);
844
845 path_ptr = fopen (path, "r+");
846 tmpf = tmpfile ();
847 tmpf_descriptor = fileno (tmpf);
848
849 // Get the key size
850 char *size_key_char = malloc (sizeof (char) * 20);
851 if (tcfs_getxattr (fuse_path, "user.key_len", size_key_char, 20) == -1)
852 {
853 perror ("Could not get file key size");
854 return -errno;
855 }
856 ssize_t size_key = strtol (size_key_char, NULL, 10);
857
858 // Retrieve the file key
859 unsigned char *encrypted_key
860 = malloc (sizeof (unsigned char) * (size_key + 1));
861 encrypted_key[size_key] = '\0';
862 if (tcfs_getxattr (fuse_path, "user.key", (char *)encrypted_key, size_key)
863 == -1)
864 {
865 perror ("Could not get file encrypted key in tcfs write");
866 return -errno;
867 }
868
869 // Decrypt the file key
870 unsigned char *decrypted_key = malloc (sizeof (unsigned char) * 33);
871 decrypted_key[32] = '\0';
872 decrypted_key = decrypt_string (encrypted_key, password);
873
874 /* Something went terribly wrong if this is the case. */
875 if (path_ptr == NULL || tmpf == NULL)
876 {
877 perror ("Something went terribly wrong, cannot create new files");
878 return -errno;
879 }
880
881 /* if the file to write to exists, read it into the tempfile */
882 if (tcfs_access (fuse_path, R_OK) == TCFS_SUCCESS && file_size (path_ptr) > 0)
883 {
884 if (do_crypt (path_ptr, tmpf, DECRYPT, decrypted_key) == false)
885 {
886 perror ("do_crypt: Cannot cypher file\n");
887 return --errno;
888 }
889 rewind (path_ptr);
890 rewind (tmpf);
891 }
892
893 /* Read our tmpfile into the buffer. */
894 res = pwrite (tmpf_descriptor, buf, size, offset);
895 if (res == -1)
896 {
897 logMessage ("%d\n", res);
898 perror ("pwrite: cannot read tmpfile into the buffer\n");
899 res = -errno;
900 }
901
902 /* Encrypt*/
903 if (do_crypt (tmpf, path_ptr, ENCRYPT, decrypted_key) == false)
904 {
905 perror ("do_crypt 2: cannot cypher file\n");
906 return -errno;
907 }
908
909 fclose (tmpf);
910 fclose (path_ptr);
911 free (encrypted_key);
912 free (decrypted_key);
913
914 return res;
915}
