109/*
110CreateSharedFile The request contains a username list and the k number for Shamir
111A new key for the file will be generated and then Shamir generates all the key-parts
112Each key-part is cyphered with the public key of the user and saved in the relative entry in the DB
113A fileID is returned in the response. This will identify the file
114*/
115func CreateSharedFile(w http.ResponseWriter, r *http.Request) {
116
117 type User struct {
118 Username string `json:"username"`
119 }
120
121 type Request struct {
122 Users []User `json:"users"`
123 K int `json:"k"`
124 }
125
126 var req Request
127
128 decoder := json.NewDecoder(r.Body)
129 err := decoder.Decode(&req)
130 if err != nil {
131 http.Error(w, err.Error(), http.StatusBadRequest)
132 return
133 }
134
135 var users []TCFSTypes.TCFSUser
136 for _, user := range req.Users {
137 tmpUser := TCFSTypes.TCFSUser{Username: user.Username}
138 err := DB.LoadUserInfoByName(&tmpUser)
139 if err != nil {
140 http.Error(w, err.Error(), http.StatusBadRequest)
141 fmt.Printf("Could not load user %v info %v\n", user, err)
142 return
143 }
144 users = append(users, tmpUser)
145 }
146
147 k := req.K
148
149 fmt.Printf("Got users and k: %v\n", k)
150
151 // Generate a new key
152 key, err := KeyTools.GenerateKey()
153 if err != nil {
154 fmt.Printf("Err: cannot not generate new key %v\n", err)
155 http.Error(w, err.Error(), http.StatusInternalServerError)
156 return
157 }
158
159 // Split the key using Shamir's secret sharing
160 shares, err := KeyTools.SplitKey(key, len(users), k)
161 if err != nil {
162 fmt.Printf("Cannot split the key %v\n", err)
163 http.Error(w, err.Error(), http.StatusInternalServerError)
164 return
165 }
166 fmt.Printf("Got %v keyparts for %v users\n", len(shares), len(users))
167
168 fileID, err := DB.GetNewFileID()
169 if err != nil {
170 fmt.Printf("Cannot generate a new fileID %v\n", err)
171 http.Error(w, err.Error(), http.StatusInternalServerError)
172 return
173 }
174 fmt.Printf("Got new file id %v\n", fileID)
175
176 var sharedFilesList []TCFSTypes.SharedFile
177 // Couple the shares with the user in the sharedFilesList
178 j := 0
179 for _, share := range shares {
180 if share == nil {
181 fmt.Printf("This share is nil\n")
182 http.Error(w, err.Error(), http.StatusInternalServerError)
183 return
184 }
185 sharedFile := TCFSTypes.SharedFile{
186 User: users[j],
187 FileID: fileID,
188 Share: share,
189 EncryptedShare: "",
190 }
191 sharedFilesList = append(sharedFilesList, sharedFile)
192 j++
193 }
194 fmt.Printf("Created %v shared files \n", len(sharedFilesList))
195
196 for _, s := range sharedFilesList {
197 fmt.Printf("%v\n", s)
198 }
199
200 err = KeyTools.EncryptSharesForSharedFileList(&sharedFilesList)
201 if err != nil {
202 fmt.Printf("Err: cannot Encrypt share list: %v\n", err)
203 http.Error(w, err.Error(), http.StatusInternalServerError)
204 return
205 }
206
207 err = DB.InsertMultipleSharedFiles(sharedFilesList)
208 if err != nil {
209 fmt.Printf("Err: cannot save list in DB %v\n", err)
210 http.Error(w, err.Error(), http.StatusInternalServerError)
211 return
212 }
213 fmt.Printf("New share saved to DB\n")
214
215 // Return a success message
216 w.WriteHeader(http.StatusOK)
217 fmt.Fprintf(w, fmt.Sprintf("fileID:%v\n", fileID))
218 fmt.Fprintf(w, "Shared file created successfully")
219}
